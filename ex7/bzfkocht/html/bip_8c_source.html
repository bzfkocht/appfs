153 static void bip_preprocess(BIP* bip)
154 {
155  bip->equs = 0;
156 
157  /* copy equations to col order and remove GE
158  */

160

162  {
163  if (bip->sense[r] == EQ)
164  {
165  bip->rhs_ord[row_cnt] = bip->rhs[r];

167  for(int c = 0; c < bip->cols; c++)

173  }

175  */
176  for(int r = 0; r < bip->rows; r++)
177  {

181  bip->rhs_ord[row_cnt] = bip->rhs[r];

183  for(int c = 0; c < bip->cols; c++)
184  bip->ac[c][row_cnt] = bip->ar[r][c];
185

209  */

558 long bip_enumerate(
559  const BIP* bip,
560  Report_fun report_sol)
561 {
562  unsigned int index32[32] =
563  {
564  0, 1, 28, 2, 29, 14, 24, 3, 30, 22, 20, 15, 25, 17, 4, 8,
565  31, 27, 13, 23, 21, 19, 16, 7, 26, 12, 18, 6, 11, 5, 10, 9
566  };
567 
568  const int cols = bip->cols;
569  const int rows = bip->rows;
570 
571  assert(bip_is_valid(bip));
572  assert(sizeof(int) == 4); //lint !e506 constant value Boolean
573 
574  /* Interesting question whether there is no or one solution.
575  */
576  if (cols == 0)
577  return 0;
578 
579  long sol_count = 0;
580  unsigned int x = 0; /* current bitvector */
581  unsigned int n = 1; /* node number */
582  unsigned int negn; /* negn = -n */
583  const double* modcol;
584  int k;
585  double r[rows];
586 
587  for(k = 0; k < rows; k++)
588  r[k] = -bip->rhs_ord[k];
589 
590  /* Check whether 0 is a feasible solution
591  */
592  sol_count += check_feasibility(bip, x, r, report_sol);
593 
594  /* Starting with x = 0000, n = 0001, negn = 1111, the algorithm
595  * enumerates all x vectors by always doing only one flip, which means
596  * we only have to update the residium for a single column.
597  * This is done as follows:
598  * 1. updatemask = n & negn (always only one bit set!)
599  * 2. modcol = bit number which is set in updatemask
600  * 2. n++, negn--
601  * 2. xor-bitflip: x ^= updatemask
602  * 3. add or substract col[modcol] from activities and check feasibility
603  * 4. as long as negn != 0, goto 1.
604  * Thereby, the x vectors are scanned in the following order:
605  * 0000, 0001, 0011, 0010, 0110, 0111, 0101, 0100,
606  * 1100, 1101, 1111, 1110, 1010, 1011, 1001, 1000
607  */
608  negn = 1u << (cols - 1);
609  negn += (negn - 1);
610 
611  while(negn != 0)
612  {
613  unsigned int updatemask = n & negn;
614 
615  /* http://en.wikipedia.org/wiki/De_Bruijn_sequence
616  */
617  unsigned int colidx = index32[(uint32_t)(updatemask * DEBRUIJN32) >> 27];
618 
619  /* ((n & negn) == (n & -n)) */
620  assert(((n + negn) & ((1u << (cols - 1)) + ((1u << (cols - 1)) - 1))) == 0);
621 
622  ++n;
623  --negn;
624  x ^= updatemask;
625 
626  modcol = bip->ac[colidx];
627 
628  if (x & updatemask)
629  {
630  for(k = 0; k < rows; k++)
631  r[k] += modcol[k];
632  }
633  else /* bit changed from 1 to 0 */
634  {
635  for(k = 0; k < rows; k++)
636  r[k] -= modcol[k];
637  }
638  sol_count += check_feasibility(bip, x, r, report_sol);
639  }
640  return sol_count;
641 }
